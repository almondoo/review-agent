+import { z } from 'zod';
+
+export const WidgetSchema = z.object({
+  id: z.string().uuid(),
+  label: z.string().min(1),
+  qty: z.number().int().nonnegative(),
+});
+export type Widget = z.infer<typeof WidgetSchema>;
+
+export function summarise(items: Widget[]): { count: number; total: number } {
+  return {
+    count: items.length,
+    total: items.reduce((sum, w) => sum + w.qty, 0),
+  };
+}
